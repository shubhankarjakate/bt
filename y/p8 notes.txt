# 👨‍💻 Full Python Code — Simple Proof-of-Work

```python id="pow6"
import hashlib
import datetime


# Block Class
class Block:

    def __init__(self, index, data, previous_hash):

        self.index = index
        self.timestamp = str(datetime.datetime.now())
        self.data = data
        self.previous_hash = previous_hash

        self.nonce = 0
        self.hash = self.calculate_hash()

    # Calculate hash
    def calculate_hash(self):

        block_data = (
            str(self.index) +
            self.timestamp +
            self.data +
            self.previous_hash +
            str(self.nonce)
        )

        return hashlib.sha256(block_data.encode()).hexdigest()

    # Mining function
    def mine_block(self, difficulty):

        print(f"\nMining Block {self.index}...")

        target = "0" * difficulty

        while self.hash[:difficulty] != target:

            self.nonce += 1
            self.hash = self.calculate_hash()

        print("Block Mined!")
        print("Nonce:", self.nonce)
        print("Hash:", self.hash)


# Blockchain Class
class Blockchain:

    def __init__(self):

        self.chain = []
        self.difficulty = 4

        # Create Genesis Block
        genesis_block = Block(0, "Genesis Block", "0")
        genesis_block.mine_block(self.difficulty)

        self.chain.append(genesis_block)

    # Get latest block
    def get_latest_block(self):

        return self.chain[-1]

    # Add block
    def add_block(self, data):

        previous_block = self.get_latest_block()

        new_block = Block(
            len(self.chain),
            data,
            previous_block.hash
        )

        new_block.mine_block(self.difficulty)

        self.chain.append(new_block)


# Create blockchain
my_blockchain = Blockchain()

# Add blocks
my_blockchain.add_block("Alice sends 5 BTC to Bob")
my_blockchain.add_block("Bob sends 2 BTC to Charlie")
my_blockchain.add_block("Charlie sends 1 BTC to David")


# Display blockchain
for block in my_blockchain.chain:

    print("\n----------------------")
    print("Block:", block.index)
    print("Data:", block.data)
    print("Nonce:", block.nonce)
    print("Hash:", block.hash)
    print("Previous Hash:", block.previous_hash)
```
